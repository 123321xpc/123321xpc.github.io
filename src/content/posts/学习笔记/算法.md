# 归并排序求逆序对

```c++
ll merge_sort(int l, int r){
	if(l >= r) return 0;
	
	int mid = l + r >> 1;
	ll res = merge_sort(l, mid) + merge_sort(mid+1, r);  //2点均在左半部分和均在右半部分
	
	//2点在两侧
	int i = l, j = mid + 1, k = 0;
	while(i<=mid && j<=r)
	{
		if(a[i] <= a[j]) t[k++] = a[i++];
		else
		{
		    //若是左侧点大于右侧点，则左侧点到mid的所有数钧与右侧点构成逆序对
			res += mid - i + 1;
			t[k++] = a[j++];
		} 
	}
	while(i<=mid) t[k++] = a[i++];
	while(j<=r) t[k++] = a[j++];
	
	for(int i=0;i<k;++i) a[l+i] = t[i];
	
	return res;
}
```



# 二分查找

## upper_bound和lower_bound

- `lower_bound`：查找数组中**第一个大于等于x**的数。
- `upper_bound`：查找数组中**第一个大于x**的数。

- **注：当数组中没有符合条件的数时，返回坐标为区间末端点，查找结果不会越过区间**

- 例：

```c++
// 查找最大的小于x的数
int fnd1(int x){
    int idx = lower_bound(a+1, a+n, x) - a; //  若全部数都小于x, 则返回n
    if(a[idx] >= x) idx--;
    return idx;
}
```

## Djakarta(邻接矩阵)

```c++
#include<bits/stdc++.h>
using namespace std;
const int N = 550, INF = 0x3f3f3f3f;
int g[N][N]; int d[N]; bool st[N];
int n, m;

int dj(){
	memset(d, INF, sizeof d);
	d[1] = 0;
	
	for(int i=1;i<=n;++i)  //循环n次
	{
		int t = -1;
		for(int j=1;j<=n;++j)  //找到最小元素
			if(!st[j] && (t == -1 || d[j] < d[t]))
				t = j;
				
		st[t] = 1; //标记已找过
		
        //更新
		for(int j=1;j<=n;++j)	d[j] = min(d[j], d[t] + g[t][j]);
	}
	
	return d[n];
}

int main(){
	cin>>n>>m;
	memset(g, INF, sizeof g);
	
	while(m--)
	{
		int a, b,c ;
		cin>>a>>b>>c;
		g[a][b] = min(g[a][b], c);
	}
	
	int t = dj();
	if(t != INF) cout<<t;
	else cout<<-1;
	return 0;
}
```



## 堆优化版Dj

```c++
#include<bits/stdc++.h>
using namespace std;
const int N = 1e5+5, INF = 0x3f3f3f3f;
typedef pair<int,int>PII;
#define x first
#define y second
vector<PII>v[N];
priority_queue<PII,vector<PII>,greater<PII>>q;
bool st[N];
int d[N];
int n, m;

int Dj(){
    memset(d, INF, sizeof d);
    d[1] = 0;
    
    q.push({0, 1});
    
    while(q.size())
    {
        auto t = q.top(); q.pop();
        int ver = t.y, dst = t.x;
        if(st[ver]) continue;
        st[ver] = 1;
        for(auto it : v[ver])
        {
            int j = it.y;
            if(d[j] > dst + it.x)
            {
                d[j] = dst + it.x;
                q.push({d[j], j});
            }
        }
    }
    
    return d[n] == INF ? -1 : d[n];
}

int main(){
    cin>>n>>m;
    while(m--)
    {
        int a, b, c;
        cin>>a>>b>>c;
        v[a].push_back({c, b});
    }
    cout<<Dj()<<endl;
    return 0;
}
```



## bellman：有边数限制的最短路

```c++
struct ed{
    int a, b, c;
}e[M];
int d[N], backup[N];
int n, m, k;

int bell(){
    memset(d, INF, sizeof d);
    d[1] = 0;

    // 循环边数k次，每次更新距离
    while(k--)
    {
        // 注意一定要用备份数组更新
        memcpy(backup, d, sizeof d);
        for(int i=0;i<m;++i)
        {
            int a = e[i].a, b = e[i].b, c = e[i].c;
            d[b] = min(d[b], backup[a] + c); //a是备份中的
        }
    }

    return d[n];
}
```

# DP

## 整数划分问题

```c++
int f[N][N]; //f[i][j]表示将i分成j份的方案数

int dp(){
    memset(f, 0, sizeof f);
    f[0][0] = 1; //初始化
    
    //划分依据：方案中的最小值为0 f[i][j-1]
    //          方案中的最小值>=1，将每个数减去1，方案数不变，共减去j f[i-j][j]
    //          故f[i][j] = f[i][j-1] + f[i-j][j]
    for(int i=0;i<=m;++i)
     for(int j=1;j<=n;++j)
     {
         f[i][j] = f[i][j-1];
         if(i >= j)  //后者只有m>=n才会有
            f[i][j] += f[i-j][j];
     }
     
    return f[m][n];
}
```



## 01背包

- 初始化：

  - 体积恰好是`j`：`f[0] = 0`，其他为`∞`

  - 体积最多为/至少为`j`：全为0

```c++
int main(){
    cin>>n>>m>>k;
    memset(f, INF, sizeof f);
    f[0][0] = 0; 
    while(k--)
    {
        int a, b, w; 
        cin>>a>>b>>w;
        for(int i=n;i>=0;--i) //当j - v < 0 时，说明第i个气缸足够满足所有氧气，只需令j-v为0即可
         for(int j=m;j>=0;--j)
          f[i][j] = min(f[i][j], f[max(0, i-a)][max(0, j-b)] + w);
    }
    
    cout<<f[n][m]<<endl;
    return 0;
}
```



### 变式：结果必须为k的倍数

```c++
int f[N][N]; //f[i][j]表示前i个中选，sum % k = j的方案中最大值
int n, k;

int main(){
    cin>>n>>k;
    memset(f, -0x3f3f3f3f, sizeof f); 
    f[0][0] = 0; //只有f[0][0]合法
    for(int i=1;i<=n;++i)
    {
        int w; cin>>w;
        for(int j=0;j<k;++j) 
            f[i][j] = max(f[i-1][j], f[i-1][((j-w)%k + k)%k] + w);
            //        由于(j-w)%k可能为负数，故需改成((j-w)%k + k)%k
    }
    
    cout<<f[n][0]<<endl; //结果：k的倍数取模为0
    return 0;
}
```





### 01背包求方案数

```c++
#include<bits/stdc++.h>
using namespace std;
const int N = 1010, INF = 0x3f3f3f3f, mod = 1e9+7;
int f[N], g[N];  //f[j] 表示体积恰好是j的方案，g[j]表示体积恰好是j的方案数
int n, m;

int main(){
    cin>>n>>m;
    memset(f, -INF, sizeof f); //初始化为-INF
    f[0] = 0; g[0] = 1; //体积为0，价值为0，方案数为1
    for(int i=0;i<n;++i)
    {
        int v, w;
        cin>>v>>w;
        for(int j=m;j>=v;j--) //与01背包相同
        {
            int cnt = 0;
            int maxv = max(f[j], f[j-v] + w); //先记录最大值从哪里来
            if(maxv == f[j]) cnt += g[j]; //不选，则方案数加上原来的g[j]
            if(maxv == f[j-v] + w) cnt = (cnt + g[j-v]) % mod; //选，则加上选的方案数，若2者值相同，则都加
            g[j] = cnt;
            f[j] = maxv;
        }
    }
    
    //找出价值最大的结果
    int res = 0;
    for(int i=1;i<=m;++i) res = max(res, f[i]);
    
    //找出价值最大结果的方案求和
    int ans = 0;
    for(int i=0;i<=m;++i)
     if(f[i] == res)
      ans = (ans + g[i]) % mod;
      
    cout<<ans<<endl;
    
    return 0;
}
```



## 背包求具体方案(字典序最小)

- 注意：求具体方案不能简化二维，且迭代必须从最后一个物品开始

### 01背包

```c++
int main(){  
    //01背包从后往前迭代，方便后面求字典序最小的方案，不能简化
    for(int i=n;i;--i)
        for(int j=0;j<=m;++j)
        {
            //注意是i+1
            f[i][j] = f[i+1][j];
            if(j >= v[i]) f[i][j] = max(f[i][j], f[i+1][j-v[i]] + w[i]);
        }
        
    //f[1][m]是最优方案的结果    
    int j = m;
    //从前往后，只要能选，必须选
    for(int i=1;i<=n;++i)
        if(j >= v[i] && f[i][j] == f[i+1][j-v[i]] + w[i])
        {
            cout<<i<<" ";
            j -= v[i]; //选完记得减去对应的体积
        } 
    return 0;
}
```

## 二进制优化多重背包

```c++
int main(){  
    for(int i=1;i<=n;++i)
    { 
        for(int k=1;k<=s[i];k<<=1) //将该物品按照二进制分为若干组
        {
            //每组进行01背包
            for(int j=m;j>=k*v[i];--j) f[j] = max(f[j], f[j-k*v[i]] + k*w[i]);

    }

    return 0;
}
```

## 分组背包求方案

```c++
#include<bits/stdc++.h>
using namespace std;
const int N = 20;
int f[N][N], w[N][N], path[N];
int n, m;

int main(){    
    for(int i=n;i;--i)  //注意从后往前
    {
        for(int k=0;k<=m;++k)   //从小到大枚举每个物品选择个数
         //在前i-1个物品中选，总体积不超过m-k的最大值，选中第i个物品
         if(f[i][m] == f[i-1][m-k] + w[i][k]) 
         {
             
             path[i] = k;  //则第i个物品选共选k
             m -= k;       //把体积减去
             break;        //一定要退出！每组只能选一次！
         }
    }
    for(int i=1;i<=n;++i) cout<<i<<" "<<path[i]<<endl;  //具体方案
    return 0;
}
```

## 树形DP

### 没有上司的舞会

```c++
#include<bits/stdc++.h>
using namespace std;
const int N = 6005;
int f[N][2]; bool st[N];
vector<int>v[N];
int w[N];
int n;

void dfs(int u){
    //初始化
    f[u][0] = 0;    //不选根
    f[u][1] = w[u]; //选根
    
    for(auto it : v[u]) //  遍历子节点
    {
        dfs(it); //计算子节点的值
        f[u][0] += max(f[it][0], f[it][1]); //不选根，子树可选择是否选根
        f[u][1] += f[it][0];                //选根，子树只能不选根
    }
}
int main(){
    cin>>n;
    for(int i=1;i<=n;++i) cin>>w[i];
    for(int i=0;i<n-1;++i)
    {
        int a, b; cin>>a>>b;
        v[b].push_back(a);

    }
    
    int root = 1;
    while(st[root]) root++;  //找出根节点
    
    dfs(root);
    
    cout<<max(f[root][0], f[root][1])<<endl;    //找出最优方案
    return 0;
```

### 树的直径

```c++
#include<bits/stdc++.h>
using namespace std;
const int N = 1e4+5;
typedef pair<int,int>PII;
#define x first
#define y second
vector<PII>v[N];
int n, ans;

int dfs(int u, int fa){
    
    int d1 = 0, d2 = 0;
    for(auto it : v[u]) //遍历所有子节点
    {
        int j = it.x, dst = it.y;
        if(j == fa) continue;  //防止搜索到父节点
        int d_max = dfs(j, u) + dst; //返回子树的最大路径+当前点的路径
        
        if(d_max >= d1) d2 = d1, d1 = d_max; //最大路径
        else
        {
            if(d_max > d2) d2 = d_max; //次大路径
        }
        
        ans = max(ans, d1 + d2); //每次比较一次
    }
    
    return d1; //返回最大路径
}

int main(){
    cin>>n;
    for(int i=0;i<n-1;++i) 
    {
        int a, b, c;
        cin>>a>>b>>c;
        v[a].push_back({b, c});
        v[b].push_back({a, c});
    }
    
    dfs(1, -1);
    
    cout<<ans<<endl;
    return 0;
}
```

### 树的中心

```c++
#include<bits/stdc++.h>
using namespace std;
const int N = 1e4+5, INF = 0x3f3f3f3f;
typedef pair<int,int>PII;
#define x first
#define y second
vector<PII>v[N];
int n;
int up[N], d1[N], s1[N], d2[N], s2[N]; 

int dfs_down(int u, int fa){ //向下走找到最长路径
    for(auto it : v[u])
    {
        int j = it.x, dst = it.y;
        if(j == fa) continue;
        int d = dfs_down(j, u) + dst;
        if(d >= d1[u])
        {
            d2[u] = d1[u]; s2[u] = s1[u];
            d1[u] = d;     s1[u] = j;
        }else{
            if(d > d2[u])
            {
                d2[u] = d;
                s2[u] = j;
            }
        }
    }
    return d1[u];
}

void dfs_up(int u, int fa){ //向上走找到最长路径
    for(auto it : v[u]) //遍历当前节点的所有子节点
    {
        int j = it.x, dst = it.y;
        if(j == fa) continue;
        //以当前的u节点为父节点，j为子节点
        if(s1[u] == j) up[j] = max(up[u], d2[u]) + dst; //若u节点的最大值是从j来的，则变成次大值
        else up[j] = max(up[u], d1[u]) + dst;
        dfs_up(j, u); //注意先操作完本层，再递归
    }
}

int main(){
    cin>>n;
    for(int i=1;i<=n-1;++i)
    {
        int a, b, c;
        cin>>a>>b>>c;
        v[a].push_back({b, c});
        v[b].push_back({a, c});
    }
    
    dfs_down(1, -1); //先向下查找
    dfs_up(1, -1);  //后向上查找
   
    int ans = INF;
    for(int i=1;i<=n;++i) ans = min(ans, max(d1[i], up[i])); //上下取最大值
    
    cout<<ans<<endl;
    return 0;
}
```

### 分治求约数之和

```c++
#include<bits/stdc++.h>
using namespace std;
const int mod = 1e9+7;
typedef long long ll;
map<ll, int>ha;
int n;

ll qmi(ll a, ll k){
    ll res = 1;
    a %= mod;
    while(k)
    {
        if(k & 1) res = res * a % mod;
        a = a * a % mod;
        k >>= 1;
    }
    
    return res;
}

ll sum(int p, int k){
    if(k == 1) return 1;
    
    if(k % 2 == 0) return (1 + qmi(p, k / 2)) * sum(p, k / 2) % mod;
    else return qmi(p, k - 1) + sum(p, k - 1) % mod;
}

int main(){
    cin>>n;
    while(n--)
    {
        ll x; cin>>x;
        for(int i=2;x>1&&i<=x/i;++i)
            while(x % i == 0)
            {
                ha[i]++;
                x /= i;
            }
        if(x > 1) ha[x]++;
    }
    
    ll res = 1;
    for(auto it : ha)   res = res * sum(it.first, it.second + 1) % mod;
    
    cout<<res<<endl;
    return 0;
}
```

# 差分约束

1. 先判断求最小值还是求最大值？
   - 最小值：求最长路，将所有不等式变为`xi >= xj + c`的形式，初始化d为负无穷
   - 最大值：求最短路，将所有不等式变为`xi <= xj + c`的形式，初始化d为正无穷
2. 建立`j -> i`，长度为`c`的边
3. 判断是否有某个点能到达所有点。
   - 若有，则以该点为原点
   - 若无，则建立虚拟原点，一般设置`x0 = 0`，然后有不等式`xi >= x0 + 1`



- 注：
  - 若建立虚拟原点后，得到的不等式是`xi >= x0`或`xi <= x0`，则可以不建立虚拟原点，在遍历时直接把所有点放入队列中即可。·

# 求矩阵中的最大最小值

```c++
#include<bits/stdc++.h>
using namespace std;
const int N = 1010, mod = 998244353;
typedef pair<int, int>PII;
typedef long long ll;
int w[N][N], maxv[N][N], minv[N][N];
int n, m, a, b;

void get_min(int g[], int dist[], int len, int k){
    deque<int>q;
    
    for(int i=1;i<=len;++i)
    {
        if(q.size() && i - q.front() >= k) q.pop_front();
        
        while(q.size() && g[q.back()] >= g[i]) q.pop_back();
        q.push_back(i);
        
        dist[i] = g[q.front()];
    }
    
    
}

void get_max(int g[], int dist[], int len, int k){
    deque<int>q;
    
    for(int i=1;i<=len;++i)
    {
        if(q.size() && i - q.front() >= k) q.pop_front();
        
        while(q.size() && g[q.back()] <= g[i]) q.pop_back();
        q.push_back(i);
        
        dist[i] = g[q.front()];
    }
}

int main(){
    cin>>n>>m>>a>>b;
    
    for(int i=1;i<=n;++i)
     for(int j=1;j<=m;++j)
      cin>>w[i][j];
      
    for(int i=1;i<=n;++i) 
    {
        get_min(w[i], minv[i], m, b);
        get_max(w[i], maxv[i], m, b);
    }
    
    ll res = 0;
    
    int t_col[N];  //取出每一列的最大值和最小值
    int t_max[N];  //暂存每个矩阵的最大值
    int t_min[N];  //暂存每个矩阵的最小值
    
    for(int i=b;i<=m;++i)
    {
        for(int j=1;j<=n;++j) t_col[j] = maxv[j][i]; //收集每一列的最大值
        get_max(t_col, t_max, n, a); //把矩形的最大值写入t_max中
        for(int j=1;j<=n;++j) t_col[j] = minv[j][i]; //收集每一列的最小值
        get_min(t_col, t_min, n, a);
        
        //计算
        for(int j=a;j<=n;++j)   res = (res + (ll)t_min[j] * t_max[j]) % mod;
    }
    
    cout<<res<<endl;
    return 0;
}
```
